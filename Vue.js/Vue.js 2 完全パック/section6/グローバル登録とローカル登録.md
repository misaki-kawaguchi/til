# グローバル登録とローカル登録

## グローバル登録

## HTML
```
<div id="app">
  <my-conponent></my-conponent>
  <my-conponent></my-conponent>
</div>
```

## JavaScript
```
Vue.component('my-conponent', {
  data: function() {
  return {
    number: 12
  }
},
template: '<p>いいね({{number}})<button @click="increment">+1</button></p>',
  methods: {
    increment: function() {
      this.number += 1;
    }
  }
})

new Vue({
  el: '#app'
})
```

## ローカル登録

## HTML
```
<div id="app">
  <my-component></my-component>
  <my-component></my-component>
</div>
```

## JavaScript
```
var component = {
  data: function() {
    return {
      number: 12
    }
  },
  template: '<p>いいね({{number}})<button @click="increment">+1</button></p>',
    methods: {
      increment: function() {
      this.number += 1;
    }
  }
}

new Vue({
  el: '#app',
  components: {
    'my-component': component
  }
})
```
